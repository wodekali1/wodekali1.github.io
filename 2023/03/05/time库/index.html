    t:=time.Now()

    t.unix()秒

    t.UnixMicro()微秒

    t.UnixMilli()毫秒

    t.UnixNano()纳秒

    测量俩个时间的差值用

    还可以用另一种方式

    用t1减去t

    begin :=time.now()

    fmt.Println(begin.Unix())

    end :=time.now()

    t3:=end.sub(begin)

    fmt.println(t3.hour())

    加法

    begin :=time.now()

    fmt.Println(begin.Unix())

    dua :=time.duration(8*time.hour)

    end:=begin.add(dua)
